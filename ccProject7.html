  function getZero(n) {
    if (isNaN(n) || n - 0 <= 0 || n - 0 > 999)return "提交的编码必须为1-999数字";
    if (n - 0 < 10) {
    return '00' + (n - 0);
    } else if (n - 0 < 100) {
    return '0' + (n - 0);
    }
  return "" + (n - 0);
  }
